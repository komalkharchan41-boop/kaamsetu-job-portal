  <script>
    function showSection(id) {
      document.querySelectorAll(".section").forEach(s => s.classList.remove("active"));
      document.getElementById(id).classList.add("active");
    }

    let generatedOTP = "";

    function sendOTP() {
      let aadhar = document.getElementById("aadhar").value;
      if (aadhar.length !== 12) {
        alert("Enter valid 12-digit Aadhaar number.");
        return;
      }
      generatedOTP = Math.floor(100000 + Math.random() * 900000).toString();
      alert("Your OTP: " + generatedOTP);
      document.getElementById("otpBox").style.display = "block";
    }

    function verifyLogin() {
      let otp = document.getElementById("otp").value;
      if (otp === generatedOTP) {
        alert("Login Successful!");
        document.getElementById("profileIcon").style.display = "block";
        showSection('profile');
      } else {
        document.getElementById("error").style.display = "block";
      }
    }

    function saveProfile() {
      const name = document.getElementById("fullName").value;
      const skills = document.getElementById("skills").value;
      const age = document.getElementById("age").value;
      const edu = document.getElementById("education").value;
      const loc = document.getElementById("location").value;
      const contact = document.getElementById("contact").value;
      const jobType = document.getElementById("jobType").value;
      const exp = document.getElementById("experience").value;
      const gmail = document.getElementById("gmail").value;
      const resume = document.getElementById("resume").value;

      if (name && skills && age && edu && loc && contact && jobType && exp && gmail) {
        const li = document.createElement("li");
        li.innerHTML = `<strong>${name}</strong> | ${age} | ${edu} | ${loc} | ${contact} | ${jobType} | ${exp} | ${gmail} | ${resume}`;
        document.getElementById("jobSeekerList").appendChild(li);
        alert("Profile saved and added to Job Seekers list!");
      } else {
        alert("Please fill all fields before saving.");
      }
    }
  </script>
